local Utility = {}
local Fonts = {
    Regular = Enum.Font.BuilderSans,
    Medium = Enum.Font.BuilderSansMedium,
    Bold = Enum.Font.BuilderSansBold,
    Code = Enum.Font.RobotoMono

function Utility:Tween(obj, props, time, style, dir)
    time = time or 0.25
    style = style or Enum.EasingStyle.Quint
    dir = dir or Enum.EasingDirection.Out
    local tween = TweenService:Create(obj, TweenInfo.new(time, style, dir), props)
    tween:Play()
    return tween
end

function Utility:Create(className, properties, children)
    local inst = Instance.new(className)
    for k, v in pairs(properties or {}) do
        if k ~= "Theme" and k ~= "Parent" then
            inst[k] = v
        end
    end
    for _, child in pairs(children or {}) do
        child.Parent = inst
    end
    if properties and properties.Parent then
        inst.Parent = properties.Parent
    end
    return inst
end

function Utility:ApplyCorner(parent, radius)
    return self:Create("UICorner", {CornerRadius = UDim.new(0, radius or 6), Parent = parent})
end

function Utility:ApplyStroke(parent, color, thickness, transparency)
    return self:Create("UIStroke", {
        Color = color or Color3.fromRGB(255, 255, 255),
        Thickness = thickness or 1,
        Transparency = transparency or 0.9,
        ApplyStrokeMode = Enum.ApplyStrokeMode.Border,
        Parent = parent
    })
end

function Utility:AddShadow(parent, intensity, size)
    return self:Create("ImageLabel", {
        Name = "Shadow",
        BackgroundTransparency = 1,
        Image = "rbxassetid://1316045217",
        ImageColor3 = Color3.fromRGB(0, 0, 0),
        ImageTransparency = intensity or 0.7,
        ScaleType = Enum.ScaleType.Slice,
        SliceCenter = Rect.new(10, 10, 118, 118),
        Size = UDim2.new(1, size or 20, 1, size or 20),
        Position = UDim2.new(0, -(size or 20)/2, 0, -(size or 20)/2),
        ZIndex = parent.ZIndex - 1,
        Parent = parent
    })
end

function Utility:MakeDraggable(dragArea, target)
    local dragging, dragInput, dragStart, startPos
    dragArea.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = target.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    dragArea.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then dragInput = input end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            Utility:Tween(target, {Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)}, 0.1)
        end
    end)
end

function Utility:GetTextBounds(text, font, size, vector)
    return TextService:GetTextSize(text, size, font, vector or Vector2.new(10000, 10000))
end

local ThemeManager = {
    Themes = {
        ModernDark = {
            Background = Color3.fromRGB(20, 21, 23),
            Surface = Color3.fromRGB(28, 29, 32),
            SurfaceHover = Color3.fromRGB(35, 36, 40),
            Border = Color3.fromRGB(45, 46, 50),
            BorderFocus = Color3.fromRGB(70, 70, 75),
            Text = Color3.fromRGB(240, 240, 240),
            TextMuted = Color3.fromRGB(150, 155, 160),
            Accent = Color3.fromRGB(59, 130, 246), -- Tailwind Blue 500
            AccentHover = Color3.fromRGB(96, 165, 250),
            Success = Color3.fromRGB(34, 197, 94),
            Warning = Color3.fromRGB(245, 158, 11),
            Danger = Color3.fromRGB(239, 68, 68),
            Info = Color3.fromRGB(56, 189, 248),
        }
    },
    Current = "ModernDark",
    Registry = {}
}

function ThemeManager:Get(key)
    return self.Themes[self.Current][key]
end

function ThemeManager:Register(instance, property, colorKey)
    if not self.Registry[instance] then self.Registry[instance] = {} end
    self.Registry[instance][property] = colorKey
    instance[property] = self:Get(colorKey)
    
    -- Cleanup destroyed instances automatically
    instance.AncestryChanged:Connect(function(_, parent)
        if not parent then
            task.delay(1, function()
                if not instance.Parent then self.Registry[instance] = nil end
            end)
        end
    end)
end

function ThemeManager:SetTheme(themeName)
    if self.Themes[themeName] then
        self.Current = themeName
        self:UpdateAll()
    end
end

function ThemeManager:UpdateAll()
    for instance, properties in pairs(self.Registry) do
        if instance and instance.Parent then
            local tweens = {}
            for prop, key in pairs(properties) do
                tweens[prop] = self:Get(key)
            end
            Utility:Tween(instance, tweens, 0.3)
        end
    end
end

-- 3. SAVE MANAGER
-- ==============================================================================
local SaveManager = {
    Folder = "AuraConfigs",
    Flags = {},
    Elements = {},
    Ignore = {}
}

function SaveManager:Init(folderName)
    self.Folder = folderName or self.Folder
    if not isfolder(self.Folder) then makefolder(self.Folder) end
    if not isfolder(self.Folder .. "/profiles") then makefolder(self.Folder .. "/profiles") end
end

function SaveManager:Save(profileName)
    local path = self.Folder .. "/profiles/" .. profileName .. ".json"
    local data = {}
    for flag, value in pairs(self.Flags) do
        if not self.Ignore[flag] then data[flag] = value end
    end
    writefile(path, HttpService:JSONEncode(data))
end

function SaveManager:Load(profileName)
    local path = self.Folder .. "/profiles/" .. profileName .. ".json"
    if isfile(path) then
        local success, data = pcall(function() return HttpService:JSONDecode(readfile(path)) end)
        if success and data then
            for flag, val in pairs(data) do
                if self.Elements[flag] then
                    self.Elements[flag]:SetValue(val)
                else
                    self.Flags[flag] = val
                end
            end
        end
    end
end

        function Tab:CreateSlider(Config)
            local flag = Config.Flag or Config.Name
            local min = Config.Min or 0
            local max = Config.Max or 100
            local default = Config.Default or min
            local decimals = Config.Decimals or 0
            SaveManager.Flags[flag] = default

            local sFrame = Utility:Create("Frame", {
                Size = UDim2.new(1, 0, 0, 64),
                Parent = Page
            })
            ThemeManager:Register(sFrame, "BackgroundColor3", "Surface")
            Utility:ApplyCorner(sFrame, 6)
            Utility:ApplyStroke(sFrame, ThemeManager:Get("Border"), 1)

            local lbl = Utility:Create("TextLabel", {
                Size = UDim2.new(1, -80, 0, 30),
                Position = UDim2.new(0, 15, 0, 5),
                BackgroundTransparency = 1,
                Text = Config.Name,
                Font = Fonts.Medium,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
                Parent = sFrame
            })
            ThemeManager:Register(lbl, "TextColor3", "Text")

            local valBoxBg = Utility:Create("Frame", {
                Size = UDim2.new(0, 45, 0, 22),
                Position = UDim2.new(1, -60, 0, 9),
                Parent = sFrame
            })
            ThemeManager:Register(valBoxBg, "BackgroundColor3", "Background")
            Utility:ApplyCorner(valBoxBg, 4)
            Utility:ApplyStroke(valBoxBg, ThemeManager:Get("BorderFocus"), 1)

            local valBox = Utility:Create("TextBox", {
                Size = UDim2.new(1, 0, 1, 0),
                BackgroundTransparency = 1,
                Text = tostring(default),
                Font = Fonts.Code,
                TextSize = 11,
                Parent = valBoxBg
            })
            ThemeManager:Register(valBox, "TextColor3", "Text")

            local trackBtn = Utility:Create("TextButton", {
                Size = UDim2.new(1, -30, 0, 20),
                Position = UDim2.new(0, 15, 0, 36),
                BackgroundTransparency = 1,
                Text = "",
                Parent = sFrame
            })

            local track = Utility:Create("Frame", {
                Size = UDim2.new(1, 0, 0, 6),
                Position = UDim2.new(0, 0, 0.5, -3),
                Parent = trackBtn
            })
            ThemeManager:Register(track, "BackgroundColor3", "Background")
            Utility:ApplyCorner(track, 3)

            local fill = Utility:Create("Frame", {
                Size = UDim2.new(0, 0, 1, 0),
                Parent = track
            })
            ThemeManager:Register(fill, "BackgroundColor3", "Accent")
            Utility:ApplyCorner(fill, 3)

            local node = Utility:Create("Frame", {
                Size = UDim2.new(0, 14, 0, 14),
                Position = UDim2.new(1, -7, 0.5, -7),
                Parent = fill
            })
            ThemeManager:Register(node, "BackgroundColor3", "Text")
            Utility:ApplyCorner(node, 100)
            Utility:AddShadow(node, 0.3, 12)

            local SliderObj = { Value = default }

            local function formatVal(v) return string.format("%."..decimals.."f", v) end

            function SliderObj:SetValue(val)
                val = math.clamp(val, min, max)
                self.Value = val
                SaveManager.Flags[flag] = val
                valBox.Text = formatVal(val)
                local pct = (val - min) / (max - min)
                Utility:Tween(fill, {Size = UDim2.new(pct, 0, 1, 0)}, 0.1)
                if Config.Callback then Config.Callback(val) end
            end

            local dragging = false
            local function updateDrag(input)
                local pct = math.clamp((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
                local val = min + ((max - min) * pct)
                if decimals == 0 then val = math.floor(val) end
                SliderObj:SetValue(val)
            end

            trackBtn.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 then
                    dragging = true
                    updateDrag(input)
                    Utility:Tween(node, {Size = UDim2.new(0, 18, 0, 18), Position = UDim2.new(1, -9, 0.5, -9)}, 0.2)
                end
            end)

            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 and dragging then
                    dragging = false
                    Utility:Tween(node, {Size = UDim2.new(0, 14, 0, 14), Position = UDim2.new(1, -7, 0.5, -7)}, 0.2)
                end
            end)

            UserInputService.InputChanged:Connect(function(input)
                if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then updateDrag(input) end
            end)

            valBox.FocusLost:Connect(function()
                local n = tonumber(valBox.Text)
                if n then SliderObj:SetValue(n) else valBox.Text = formatVal(SliderObj.Value) end
            end)

            SaveManager.Elements[flag] = SliderObj
            SliderObj:SetValue(default)

            return SliderObj
        end

        function Tab:CreateDropdown(Config)
            local flag = Config.Flag or Config.Name
            local options = Config.Options or {}
            local default = Config.Default or options[1]
            SaveManager.Flags[flag] = default

            local dFrame = Utility:Create("Frame", {
                Size = UDim2.new(1, 0, 0, 65),

            })
            ThemeManager:Register(dFrame, "BackgroundColor3", "Surface")

                BackgroundTransparency = 1,

                Font = Fonts.Regular,
                TextSize = 12,
                TextXAlignment = Enum.TextXAlignment.Left,
                Parent = box
            })
            ThemeManager:Register(selLbl, "TextColor3", "Text")

            local icon = Utility:Create("ImageLabel", {
                Size = UDim2.new(0, 14, 0, 14),
                Position = UDim2.new(1, -20, 0.5, -7),
                BackgroundTransparency = 1,
                Image = "rbxassetid://10888331510", -- Arrow Down
                Rotation = 90,
                Parent = box
            })
            ThemeManager:Register(icon, "ImageColor3", "TextMuted")

            -- Dropdown Menu (Parented to Overlay to prevent clipping)
            local menu = Utility:Create("ScrollingFrame", {
                Size = UDim2.new(0, box.AbsoluteSize.X, 0, 0),
                Position = UDim2.new(0, box.AbsolutePosition.X, 0, box.AbsolutePosition.Y + box.AbsoluteSize.Y + 5),
                ScrollBarThickness = 2,
                BorderSizePixel = 0,
                ZIndex = 5001,
                Visible = false,
                Parent = Window.Overlay
            })
            ThemeManager:Register(menu, "BackgroundColor3", "SurfaceHover")
            ThemeManager:Register(menu, "ScrollBarImageColor3", "BorderFocus")
            Utility:ApplyCorner(menu, 6)
            Utility:ApplyStroke(menu, ThemeManager:Get("Border"), 1)
            Utility:AddShadow(menu, 0.4)

            local listLayout = Utility:Create("UIListLayout", {
                Padding = UDim.new(0, 2),
                HorizontalAlignment = Enum.HorizontalAlignment.Center,
                Parent = menu
            })

            local DropdownObj = { Value = default, Options = options, IsOpen = false }

            local updater
            function DropdownObj:Toggle()
                self.IsOpen = not self.IsOpen
                if self.IsOpen then
                    menu.Position = UDim2.new(0, box.AbsolutePosition.X, 0, box.AbsolutePosition.Y + box.AbsoluteSize.Y + 5)
                    menu.Size = UDim2.new(0, box.AbsoluteSize.X, 0, 0)
                    menu.Visible = true
                    local targetY = math.min(#self.Options * 30 + 4, 150)
                    Utility:Tween(menu, {Size = UDim2.new(0, box.AbsoluteSize.X, 0, targetY)}, 0.2)
                    Utility:Tween(icon, {Rotation = -90}, 0.2)
                    
                    -- Track position strictly when open
                    updater = RunService.RenderStepped:Connect(function()
                        menu.Position = UDim2.new(0, box.AbsolutePosition.X, 0, box.AbsolutePosition.Y + box.AbsoluteSize.Y + 5)
                        menu.Size = UDim2.new(0, box.AbsoluteSize.X, menu.Size.Y.Offset)
                    end)
                else
                    if updater then updater:Disconnect() end
                    Utility:Tween(menu, {Size = UDim2.new(0, box.AbsoluteSize.X, 0, 0)}, 0.2)
                    Utility:Tween(icon, {Rotation = 90}, 0.2)
                    task.delay(0.2, function() if not self.IsOpen then menu.Visible = false end end)
                end
            end

            box.MouseButton1Click:Connect(function() DropdownObj:Toggle() end)

            function DropdownObj:SetValue(val)
                self.Value = val
                SaveManager.Flags[flag] = val
                selLbl.Text = tostring(val)
                if Config.Callback then Config.Callback(val) end
            end

            function DropdownObj:Refresh(newOptions)
                self.Options = newOptions or {}
                for _, child in pairs(menu:GetChildren()) do
                    if child:IsA("TextButton") then child:Destroy() end
                end

                for _, opt in pairs(self.Options) do
                    local optBtn = Utility:Create("TextButton", {
                        Size = UDim2.new(1, -8, 0, 28),
                        BackgroundTransparency = 1,
                        Text = "",
                        Parent = menu
                    })
                    local optLbl = Utility:Create("TextLabel", {
                        Size = UDim2.new(1, -20, 1, 0),
                        Position = UDim2.new(0, 10, 0, 0),
                        BackgroundTransparency = 1,
                        Text = tostring(opt),
                        Font = Fonts.Regular,
                        TextSize = 12,
                        TextXAlignment = Enum.TextXAlignment.Left,
                        Parent = optBtn
                    })
                    ThemeManager:Register(optLbl, "TextColor3", "TextMuted")
                    
                    local bg = Utility:Create("Frame", {
                        Size = UDim2.new(1, 0, 1, 0),
                        BackgroundTransparency = 1,
                        Parent = optBtn
                    })
                    ThemeManager:Register(bg, "BackgroundColor3", "Accent")
                    Utility:ApplyCorner(bg, 4)
                    bg.ZIndex = optBtn.ZIndex - 1

                    optBtn.MouseEnter:Connect(function()
                        Utility:Tween(bg, {BackgroundTransparency = 0.8}, 0.1)
                        Utility:Tween(optLbl, {TextColor3 = ThemeManager:Get("Text")}, 0.1)
                    end)
                    optBtn.MouseLeave:Connect(function()
                        Utility:Tween(bg, {BackgroundTransparency = 1}, 0.1)
                        Utility:Tween(optLbl, {TextColor3 = ThemeManager:Get("TextMuted")}, 0.1)
                    end)
                    optBtn.MouseButton1Click:Connect(function()
                        self:SetValue(opt)
                        self:Toggle()
                    end)
                end
                menu.CanvasSize = UDim2.new(0, 0, 0, listLayout.AbsoluteContentSize.Y + 4)
                if self.IsOpen then
                    local targetY = math.min(#self.Options * 30 + 4, 150)
                    Utility:Tween(menu, {Size = UDim2.new(0, box.AbsoluteSize.X, 0, targetY)}, 0.2)
                end
            end

            SaveManager.Elements[flag] = DropdownObj
            DropdownObj:Refresh(options)
            if default then DropdownObj:SetValue(default) end

            return DropdownObj
        end
