local Utility = {}

local _uidCounter = 0
function Utility:UUID()
    _uidCounter = _uidCounter + 1
    return "cac_" .. _uidCounter
end

function Utility:Create(className, properties, children)
    local inst = Instance.new(className)
    for k, v in pairs(properties or {}) do
        if k == "Theme" then
            for prop, key in pairs(v) do
                ThemeManager:Register(inst, prop, key)
            end
        elseif k ~= "Parent" then
            inst[k] = v
        end
    end
    for _, child in pairs(children or {}) do
        child.Parent = inst
    end
    if properties and properties.Parent then
        inst.Parent = properties.Parent
    end
    return inst
end

function Utility:Tween(obj, props, duration, style, dir)
    if not obj or not obj.Parent then return end
    duration = duration or 0.2
    if AnimManager and AnimManager.ReducedMotion then duration = 0.05 end
    style = style or Enum.EasingStyle.Quart
    dir = dir or Enum.EasingDirection.Out
    local tween = TweenService:Create(obj, TweenInfo.new(duration, style, dir), props)
    tween:Play()
    return tween
end

function Utility:ApplyCorner(parent, radius)
    return self:Create("UICorner", {CornerRadius = UDim.new(0, radius or 6), Parent = parent})
end

function Utility:ApplyStroke(parent, colorKey, thickness, mode)
    return self:Create("UIStroke", {
        Thickness = thickness or 1,
        ApplyStrokeMode = mode or Enum.ApplyStrokeMode.Border,
        Theme = {Color = colorKey or "Border"},
        Parent = parent
    })
end

function Utility:ApplyPadding(parent, top, right, bottom, left)
    return self:Create("UIPadding", {
        PaddingTop = UDim.new(0, top or 0),
        PaddingRight = UDim.new(0, right or top or 0),
        PaddingBottom = UDim.new(0, bottom or top or 0),
        PaddingLeft = UDim.new(0, left or right or top or 0),
        Parent = parent
    })
end

function Utility:AddShadow(parent, intensity)
    return self:Create("ImageLabel", {
        Name = "Shadow",
        BackgroundTransparency = 1,
        Image = "rbxassetid://1316045217",
        ImageColor3 = Color3.fromRGB(0, 0, 0),
        ImageTransparency = intensity or 0.6,
        ScaleType = Enum.ScaleType.Slice,
        SliceCenter = Rect.new(10, 10, 118, 118),
        Size = UDim2.new(1, 24, 1, 24),
        Position = UDim2.new(0, -12, 0, -12),
        ZIndex = -1,
        Parent = parent
    })
end

function Utility:MakeDraggable(dragArea, target)
    local dragging, dragInput, dragStart, startPos
    dragArea.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = target.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    dragArea.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then dragInput = input end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            Utility:Tween(target, {
                Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            }, 0.08)
        end
    end)
end

function Utility:GetTextBounds(text, font, size, vector)
    local ok, result = pcall(function()
        return TextService:GetTextSize(text, size, font, vector or Vector2.new(10000, 10000))
    end)
    return ok and result or Vector2.new(100, 20)
end

function Utility:Ripple(button, parent)
    local ripple = self:Create("Frame", {
        Size = UDim2.new(0, 0, 0, 0),
        Position = UDim2.new(0.5, 0, 0.5, 0),
        AnchorPoint = Vector2.new(0.5, 0.5),
        BackgroundColor3 = Color3.fromRGB(255, 255, 255),
        BackgroundTransparency = 0.85,
        Parent = parent or button
    })
    self:ApplyCorner(ripple, 1000)
    self:Tween(ripple, {Size = UDim2.new(2, 0, 2, 0), BackgroundTransparency = 1}, 0.4)
    task.delay(0.45, function()
        if ripple and ripple.Parent then ripple:Destroy() end
    end)
end

function ThemeManager:Get(key)
    local theme = self.Themes[self.Current]
    return theme and theme[key] or Color3.fromRGB(255, 0, 255)
end

function ThemeManager:Register(instance, property, colorKey)
    if not self.Registry[instance] then self.Registry[instance] = {} end
    self.Registry[instance][property] = colorKey
    pcall(function() instance[property] = self:Get(colorKey) end)
end

function ThemeManager:Unregister(instance)
    self.Registry[instance] = nil
end

function ThemeManager:SetTheme(name)
    if not self.Themes[name] then return end
    self.Current = name
    self:UpdateAll()
end

function ThemeManager:SetAccent(color)
    for _, theme in pairs(self.Themes) do
        theme.Accent = color
        theme.AccentSoft = color
    end
    self:UpdateAll()
end

function ThemeManager:UpdateAll()
    local duration = (AnimManager and AnimManager.ReducedMotion) and 0.05 or 0.3
    for instance, properties in pairs(self.Registry) do
        if instance and instance.Parent then
            for prop, key in pairs(properties) do
                pcall(function()
                    TweenService:Create(instance, TweenInfo.new(duration, Enum.EasingStyle.Quart), {[prop] = self:Get(key)}):Play()
                end)
            end
        else
            self.Registry[instance] = nil
        end
    end
end

function ThemeManager:AddTheme(name, tokens)
    self.Themes[name] = tokens
end

local SaveManager = {
    Folder = "CAC_Ultimate",
    Flags = {},
    Options = {},
    Ignore = {},
    AutoSave = false,
    AutoLoad = false,
    DefaultProfile = "default",
    _hooks = {},
}

function SaveManager:Init(folderName)
    self.Folder = folderName or self.Folder
    if not isfolder(self.Folder) then makefolder(self.Folder) end
    if not isfolder(self.Folder .. "/configs") then makefolder(self.Folder .. "/configs") end
end

function SaveManager:GetPath(name)
    return self.Folder .. "/configs/" .. name .. ".json"
end

function SaveManager:Save(name)
    name = name or self.DefaultProfile
    local data = {}
    for flag, _ in pairs(self.Options) do
        if not self.Ignore[flag] then
            data[flag] = self.Flags[flag]
        end
    end
    pcall(function() writefile(self:GetPath(name), HttpService:JSONEncode(data)) end)
end

function SaveManager:Load(name)
    name = name or self.DefaultProfile
    local path = self:GetPath(name)
    if not isfile(path) then return false end
    local ok, data = pcall(function() return HttpService:JSONDecode(readfile(path)) end)
    if ok and data then
        for flag, val in pairs(data) do
            if self.Options[flag] then
                pcall(function() self.Options[flag]:SetValue(val) end)
            end
            self.Flags[flag] = val
        end
        self:RunHooks()
        return true
    end
    return false
end

function SaveManager:Delete(name)
    local path = self:GetPath(name)
    if isfile(path) then pcall(function() delfile(path) end) end
end

function SaveManager:Overwrite(name)
    self:Save(name)
end

function SaveManager:ListProfiles()
    local profiles = {}
    local ok, files = pcall(function() return listfiles(self.Folder .. "/configs") end)
    if ok and files then
        for _, file in ipairs(files) do
            local name = file:match("([^/\\]+)%.json$")
            if name then table.insert(profiles, name) end
        end
    end
    return profiles
end

function SaveManager:ExportJSON(name)
    name = name or self.DefaultProfile
    local path = self:GetPath(name)
    if isfile(path) then
        return pcall(function() return readfile(path) end)
    end
    return false, "{}"
end

function SaveManager:ImportJSON(jsonStr, profileName)
    profileName = profileName or "imported"
    local ok, data = pcall(function() return HttpService:JSONDecode(jsonStr) end)
    if ok and data then
        pcall(function() writefile(self:GetPath(profileName), HttpService:JSONEncode(data)) end)
        return true
    end
    return false
end

function SaveManager:ResetConfig()
    for flag, option in pairs(self.Options) do
        if option._default ~= nil then
            pcall(function() option:SetValue(option._default) end)
        end
    end
end

function SaveManager:SetIgnored(flag, ignored)
    self.Ignore[flag] = ignored or nil
end

function SaveManager:AddHook(fn)
    table.insert(self._hooks, fn)
end

function SaveManager:RunHooks()
    for _, fn in ipairs(self._hooks) do
        pcall(fn)
    end
end

function SaveManager:EnableAutoSave(interval)
    self.AutoSave = true
    task.spawn(function()
        while self.AutoSave do
            task.wait(interval or 30)
            if self.AutoSave then self:Save(self.DefaultProfile) end
        end
    end)
end

            TextXAlignment = Enum.TextXAlignment.Left,
            Theme = {TextColor3 = "Text"},
            ZIndex = 161,
            Parent = modalCard,
        })

        -- Content
        if config.Content then
            Utility:Create("TextLabel", {

                BackgroundTransparency = 1,
                Text = config.Content,

            })
        end

        local function closeModal()
            Utility:Tween(modalCard, {Size = UDim2.new(0, config.Width or 380, 0, 0)}, 0.25)
            Utility:Tween(ModalOverlay, {BackgroundTransparency = 1}, 0.25)
            task.delay(0.3, function()
                pcall(function() modalCard:Destroy() end)

            Parent = ModalOverlay,

        overlayBtn.MouseButton1Click:Connect(function()
            closeModal()
            overlayBtn:Destroy()
        end)

        return {Card = modalCard, Close = closeModal}
    end

        Tab.Icon = Icon
        Tab.Indicator = Indicator

            if Window.CurrentTab ~= Tab then

            if Window.CurrentTab ~= Tab then
                Utility:Tween(BtnBg, {BackgroundTransparency = 1}, 0.15)
            end
        end)

        function Tab:Show()
            if Window.CurrentTab and Window.CurrentTab ~= self then
                local old = Window.CurrentTab
                old.Page.Visible = false
                Utility:Tween(old.BtnBg, {BackgroundTransparency = 1}, 0.2)
                Utility:Tween(old.Label, {TextColor3 = ThemeManager:Get("TextMuted")}, 0.2)
                Utility:Tween(old.Label, {Font = Enum.Font.GothamMedium}, 0)
                if old.Icon then Utility:Tween(old.Icon, {ImageColor3 = ThemeManager:Get("TextMuted")}, 0.2) end
                Utility:Tween(old.Indicator, {Size = UDim2.new(0, 3, 0, 0)}, 0.2)
            end
            Window.CurrentTab = self
            Page.Visible = true
            Page.CanvasPosition = Vector2.new(0, 0)
            Utility:Tween(BtnBg, {BackgroundTransparency = 0}, 0.2)
            Utility:Tween(Label, {TextColor3 = ThemeManager:Get("Text")}, 0.2)
            Label.Font = Enum.Font.GothamBold
            if iconId then Utility:Tween(Icon, {ImageColor3 = ThemeManager:Get("Accent")}, 0.2) end
            Utility:Tween(Indicator, {Size = UDim2.new(0, 3, 0, 14)}, 0.25, Enum.EasingStyle.Back)
        end

        Btn.MouseButton1Click:Connect(function() Tab:Show() end)
        if not Window.CurrentTab then Tab:Show() end
        table.insert(Window.Tabs, Tab)

        function Tab:CreateParagraph(cfg)
            cfg = cfg or {}
            local title = cfg.Title or ""
            local content = cfg.Content or ""
            local bounds = Utility:GetTextBounds(content, Enum.Font.GothamMedium, 11, Vector2.new(540, 2000))
            local totalH = (title ~= "" and 24 or 0) + bounds.Y + 20

            local f = Utility:Create("Frame", {
                Size = UDim2.new(1, 0, 0, totalH),
                Theme = {BackgroundColor3 = "Surface"},
                LayoutOrder = nextOrder(),
                Parent = Page,
            })
            Utility:ApplyCorner(f, 6)
            Utility:ApplyStroke(f, "Border")

            local yOff = 10
            local titleLbl
            if title ~= "" then
                titleLbl = Utility:Create("TextLabel", {
                    Size = UDim2.new(1, -20, 0, 18),
                    Position = UDim2.new(0, 10, 0, yOff),
                    BackgroundTransparency = 1,
                    Text = title,
                    Font = Enum.Font.GothamBold,
                    TextSize = 12,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    Theme = {TextColor3 = "Text"},
                    Parent = f,
                })
                yOff = yOff + 22
            end

            local contentLbl = Utility:Create("TextLabel", {
                Size = UDim2.new(1, -20, 0, bounds.Y + 4),
                Position = UDim2.new(0, 10, 0, yOff),
                BackgroundTransparency = 1,
                Text = content,
                Font = Enum.Font.GothamMedium,

                TextWrapped = true,
                TextXAlignment = Enum.TextXAlignment.Left,
                TextYAlignment = Enum.TextYAlignment.Top,

            function obj:SetTitle(t) if titleLbl then titleLbl.Text = t end end

            function obj:Destroy() f:Destroy() end
            return obj
        end

        -- Slider
        function Tab:CreateSlider(cfg)
            cfg = cfg or {}
            local flag = cfg.Flag or cfg.Name or Utility:UUID()
            local min = cfg.Min or 0
            local max = cfg.Max or 100
            local default = math.clamp(cfg.Default or min, min, max)
            local decimals = cfg.Decimals or 0
            local suffix = cfg.Suffix or ""
            SaveManager.Flags[flag] = default

            local f = Utility:Create("Frame", {
                Size = UDim2.new(1, 0, 0, AnimManager:ElementHeight(52)),
                Theme = {BackgroundColor3 = "Surface"},

                Parent = Page,
            })
            Utility:ApplyCorner(f, 6)
            Utility:ApplyStroke(f, "Border")

            Utility:Create("TextLabel", {
                Size = UDim2.new(0.6, 0, 0, 20),
                Position = UDim2.new(0, 12, 0, 6),
                BackgroundTransparency = 1,
                Text = cfg.Name or "Slider",
                Font = Enum.Font.GothamMedium,
                TextSize = 12,
                TextXAlignment = Enum.TextXAlignment.Left,
                Theme = {TextColor3 = "Text"},
                Parent = f,
            })

            local valLabel = Utility:Create("TextLabel", {
                Size = UDim2.new(0.35, 0, 0, 20),
                Position = UDim2.new(0.62, 0, 0, 6),
                BackgroundTransparency = 1,
                Text = string.format("%."..decimals.."f", default) .. suffix,
                Font = Enum.Font.Code,
                TextSize = 11,
                TextXAlignment = Enum.TextXAlignment.Right,
                Theme = {TextColor3 = "Accent"},
                Parent = f,
            })

            local trackBtn = Utility:Create("TextButton", {
                Size = UDim2.new(1, -24, 0, 18),
                Position = UDim2.new(0, 12, 0, 30),
                BackgroundTransparency = 1,
                Text = "",
                Parent = f,
            })

            local track = Utility:Create("Frame", {
                Size = UDim2.new(1, 0, 0, 4),
                Position = UDim2.new(0, 0, 0.5, -2),
                Theme = {BackgroundColor3 = "SurfaceAlt"},
                Parent = trackBtn,
            })
            Utility:ApplyCorner(track, 2)

            local initPct = (default - min) / math.max(max - min, 1)
            local fill = Utility:Create("Frame", {
                Size = UDim2.new(initPct, 0, 1, 0),
                Theme = {BackgroundColor3 = "Accent"},
                Parent = track,
            })
            Utility:ApplyCorner(fill, 2)

            local knob = Utility:Create("Frame", {
                Size = UDim2.new(0, 12, 0, 12),
                Position = UDim2.new(1, -6, 0.5, -6),
                Theme = {BackgroundColor3 = "Text"},
                ZIndex = 2,
                Parent = fill,
            })
            Utility:ApplyCorner(knob, 100)

            local obj = {Value = default, _default = default, _disabled = false}
            local dragging = false

            local function fmt(v) return string.format("%."..decimals.."f", v) .. suffix end

            function obj:SetValue(val)
                val = math.clamp(val, min, max)
                if decimals == 0 then val = math.floor(val + 0.5) end
                self.Value = val
                SaveManager.Flags[flag] = val
                valLabel.Text = fmt(val)
                local pct = (val - min) / math.max(max - min, 1)
                Utility:Tween(fill, {Size = UDim2.new(pct, 0, 1, 0)}, 0.08)
                if cfg.Callback then pcall(cfg.Callback, val) end
            end

            function obj:GetValue() return self.Value end
            function obj:SetVisible(v) f.Visible = v end
            function obj:SetDisabled(d) self._disabled = d; f.BackgroundTransparency = d and 0.5 or 0 end
            function obj:Destroy() f:Destroy() end

            local function updateDrag(input)
                local pct = math.clamp((input.Position.X - track.AbsolutePosition.X) / math.max(track.AbsoluteSize.X, 1), 0, 1)
                local val = min + (max - min) * pct
                obj:SetValue(val)
            end

            trackBtn.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 and not obj._disabled then
                    dragging = true
                    updateDrag(input)
                    Utility:Tween(knob, {Size = UDim2.new(0, 16, 0, 16), Position = UDim2.new(1, -8, 0.5, -8)}, 0.15)
                end
            end)
            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 and dragging then
                    dragging = false
                    Utility:Tween(knob, {Size = UDim2.new(0, 12, 0, 12), Position = UDim2.new(1, -6, 0.5, -6)}, 0.15)
                end
            end)
            UserInputService.InputChanged:Connect(function(input)
                if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                    updateDrag(input)
                end
            end)

            SaveManager.Options[flag] = obj
            obj:SetValue(default)
            return obj
        end

                BackgroundTransparency = 1,
                ClipsDescendants = true,
                Parent = f,
            })
            local optLayout = Utility:Create("UIListLayout", {
                Padding = UDim.new(0, 2),
                SortOrder = Enum.SortOrder.LayoutOrder,
                Parent = optionContainer,
            })

            local obj = {Value = default, _default = default, _disabled = false, _optionBtns = {}}

            local function buildOptions()
                for _, c in ipairs(optionContainer:GetChildren()) do
                    if c:IsA("TextButton") then c:Destroy() end
                end
                obj._optionBtns = {}
                for i, opt in ipairs(options) do
                    local oBtn = Utility:Create("TextButton", {
                        Size = UDim2.new(1, 0, 0, 28),
                        Text = opt,

                        Utility:Tween(oBtn, {BackgroundColor3 = ThemeManager:Get("SurfaceAlt")}, 0.1)

                        obj:SetValue(opt)
                        obj:Close()
                    end)
                    table.insert(obj._optionBtns, oBtn)
                end
            end

                if obj._disabled then return end
                isOpen = true
                local listH = math.min(#options * 30, 150)
                local totalH = 38 + listH + 8
                optionContainer.Size = UDim2.new(1, -16, 0, listH)
                Utility:Tween(f, {Size = UDim2.new(1, 0, 0, totalH)}, 0.25, Enum.EasingStyle.Back)

                selectedLabel.Text = val or "Select..."
                selectedLabel.TextColor3 = ThemeManager:Get(val and "Accent" or "TextSoft")
                for _, b in ipairs(self._optionBtns) do
                    b.TextColor3 = ThemeManager:Get(b.Text == val and "Accent" or "TextMuted")
                end
                if cfg.Callback then pcall(cfg.Callback, val) end
            end

            function obj:SetOptions(newOpts)
                options = newOpts
                buildOptions()
            end

            function obj:AddOption(opt)
                table.insert(options, opt)
                buildOptions()
            end

            function obj:RemoveOption(opt)
                for i, v in ipairs(options) do
                    if v == opt then table.remove(options, i); break end
                end
                buildOptions()
            end

            function obj:Refresh() buildOptions() end
            function obj:GetValue() return self.Value end
            function obj:SetVisible(v) f.Visible = v end
            function obj:SetDisabled(d) self._disabled = d; f.BackgroundTransparency = d and 0.5 or 0 end
            function obj:Destroy() f:Destroy() end

            headerBtn.MouseButton1Click:Connect(function()
                if isOpen then obj:Close() else obj:Open() end
            end)

            buildOptions()
            SaveManager.Options[flag] = obj
            return obj
        end

                Parent = optContainer,
            })

            local obj = {Value = default, _default = default, _disabled = false}

            local function getList()
                local r = {}
                for _, v in ipairs(options) do if selected[v] then table.insert(r, v) end end
                return r
            end

            local function updateDisplay()
                selLabel.Text = getDisplayText()
                selLabel.TextColor3 = ThemeManager:Get(next(selected) and "Accent" or "TextSoft")
                obj.Value = getList()
                SaveManager.Flags[flag] = obj.Value
            end

            local function buildOptions()
                for _, c in ipairs(optContainer:GetChildren()) do
                    if c:IsA("TextButton") then c:Destroy() end
                end
                for i, opt in ipairs(options) do
                    local oBtn = Utility:Create("TextButton", {
                        Size = UDim2.new(1, 0, 0, 28),
                        Text = (selected[opt] and "✓ " or "   ") .. opt,
                        Font = Enum.Font.GothamMedium,
                        TextSize = 11,
                        TextXAlignment = Enum.TextXAlignment.Left,
                        Theme = {BackgroundColor3 = "SurfaceAlt", TextColor3 = selected[opt] and "Accent" or "TextMuted"},
                        LayoutOrder = i,
                        Parent = optContainer,
                    })
                    Utility:ApplyCorner(oBtn, 4)
                    Utility:ApplyPadding(oBtn, 0, 0, 0, 8)

                    oBtn.MouseButton1Click:Connect(function()
                        selected[opt] = not selected[opt] or nil
                        oBtn.Text = (selected[opt] and "✓ " or "   ") .. opt
                        oBtn.TextColor3 = ThemeManager:Get(selected[opt] and "Accent" or "TextMuted")
                        updateDisplay()
                        if cfg.Callback then pcall(cfg.Callback, getList()) end
                    end)
                end
            end

            headerBtn.MouseButton1Click:Connect(function()
                if obj._disabled then return end
                isOpen = not isOpen
                if isOpen then
                    local listH = math.min(#options * 30, 150)
                    optContainer.Size = UDim2.new(1, -16, 0, listH)
                    Utility:Tween(f, {Size = UDim2.new(1, 0, 0, 38 + listH + 8)}, 0.25, Enum.EasingStyle.Back)
                    chevron.Text = "▴"
                else
                    Utility:Tween(f, {Size = UDim2.new(1, 0, 0, AnimManager:ElementHeight(38))}, 0.2)
                    chevron.Text = "▾"
                end
            end)

            function obj:SetValue(val)
                selected = {}
                if type(val) == "table" then for _, v in ipairs(val) do selected[v] = true end end
                updateDisplay()
                buildOptions()
            end
            function obj:SetOptions(newOpts) options = newOpts; buildOptions(); updateDisplay() end
            function obj:GetValue() return getList() end
            function obj:SetVisible(v) f.Visible = v end
            function obj:SetDisabled(d) self._disabled = d; f.BackgroundTransparency = d and 0.5 or 0 end
            function obj:Destroy() f:Destroy() end

            buildOptions()
            SaveManager.Options[flag] = obj
            return obj
        end

                Theme = {TextColor3 = "Text"},
                Parent = f,
            })

            local keyBtn = Utility:Create("TextButton", {
                Size = UDim2.new(0, 80, 0, 24),
                Position = UDim2.new(1, -92, 0.5, -12),
                Text = default == Enum.KeyCode.Unknown and "None" or default.Name,
                Font = Enum.Font.Code,
                TextSize = 11,
                Theme = {BackgroundColor3 = "SurfaceAlt", TextColor3 = "TextMuted"},
                Parent = f,
            })
            Utility:ApplyCorner(keyBtn, 4)
            Utility:ApplyStroke(keyBtn, "Border")

            local obj = {Value = default, _default = default}

            function obj:SetValue(val)
                if type(val) == "string" then
                    val = Enum.KeyCode[val] or Enum.KeyCode.Unknown
                end
                self.Value = val
                SaveManager.Flags[flag] = val.Name
                keyBtn.Text = val == Enum.KeyCode.Unknown and "None" or val.Name
                if cfg.Callback then pcall(cfg.Callback, val) end
            end

            keyBtn.MouseButton1Click:Connect(function()
                if listening then return end
                listening = true
                keyBtn.Text = "..."
                keyBtn.TextColor3 = ThemeManager:Get("Accent")
                local conn
                conn = UserInputService.InputBegan:Connect(function(input, gpe)
                    if input.UserInputType == Enum.UserInputType.Keyboard then
                        listening = false
                        conn:Disconnect()
                        local key = input.KeyCode
                        if key == Enum.KeyCode.Escape then key = Enum.KeyCode.Unknown end
                        obj:SetValue(key)
                        keyBtn.TextColor3 = ThemeManager:Get("TextMuted")
                    end
                end)
            end)

            -- Fire callback on key press
            UserInputService.InputBegan:Connect(function(input, gpe)
                if gpe or listening then return end
                if input.KeyCode == obj.Value and obj.Value ~= Enum.KeyCode.Unknown then
                    if cfg.OnPress then pcall(cfg.OnPress) end
                end
            end)

            function obj:GetValue() return self.Value end
            function obj:SetVisible(v) f.Visible = v end
            function obj:Destroy() f:Destroy() end

            SaveManager.Options[flag] = obj
            return obj
        end

                Position = UDim2.new(0, 8, 0, 0),
                BackgroundTransparency = 1,
                Text = "⌕",
                Font = Enum.Font.GothamBold,
                TextSize = 14,
                Theme = {TextColor3 = "TextSoft"},
                Parent = f,
            })

            local box = Utility:Create("TextBox", {
                Size = UDim2.new(1, -36, 1, 0),
                Position = UDim2.new(0, 28, 0, 0),
                BackgroundTransparency = 1,
                PlaceholderText = cfg.Placeholder or "Search elements...",
                Text = "",
                Font = Enum.Font.GothamMedium,
                TextSize = 12,
                TextXAlignment = Enum.TextXAlignment.Left,
                ClearTextOnFocus = false,
                Theme = {TextColor3 = "Text"},
                Parent = f,
            })

            box:GetPropertyChangedSignal("Text"):Connect(function()
                local query = box.Text:lower()
                for _, child in ipairs(Page:GetChildren()) do
                    if child:IsA("Frame") and child ~= f then
                        local shouldShow = query == ""
                        if not shouldShow then
                            for _, desc in ipairs(child:GetDescendants()) do
                                if desc:IsA("TextLabel") and desc.Text:lower():find(query, 1, true) then
                                    shouldShow = true
                                    break
                                end
                            end
                        end
                        child.Visible = shouldShow
                    end
                end
            end)

            local obj = {}
            function obj:SetVisible(v) f.Visible = v end
            function obj:Destroy() f:Destroy() end
            return obj
        end

                Theme = {BackgroundColor3 = "Background"},

                BackgroundTransparency = 1,
                Text = ">",
                Font = Enum.Font.Code,
                TextSize = 14,
                Theme = {TextColor3 = "Accent"},
                Parent = f,
            })

            local box = Utility:Create("TextBox", {
                Size = UDim2.new(1, -32, 1, 0),
                Position = UDim2.new(0, 24, 0, 0),
                BackgroundTransparency = 1,
                PlaceholderText = cfg.Placeholder or "Enter command...",
                Text = "",
                Font = Enum.Font.Code,
                TextSize = 12,
                TextXAlignment = Enum.TextXAlignment.Left,
                ClearTextOnFocus = false,
                Theme = {TextColor3 = "Text"},
                Parent = f,
            })

            box.FocusLost:Connect(function(enter)
                if enter then
                    local cmd = box.Text:lower():match("^%s*(.-)%s*$")
                    if commands[cmd] then
                        pcall(commands[cmd])
                    elseif cfg.Callback then
                        pcall(cfg.Callback, box.Text)
                    end
                    box.Text = ""
                end
            end)

            local obj = {}
            function obj:AddCommand(name, fn) commands[name:lower()] = fn end
            function obj:SetVisible(v) f.Visible = v end
            function obj:Destroy() f:Destroy() end
            return obj
        end

    -- BUILT-IN SETTINGS TAB HELPER

    function Window:CreateSettingsTab(name)
        local tab = Window:CreateTab(name or "Settings")

        -- Theme Section
        tab:CreateSection("Appearance")

        tab:CreateDropdown({
            Name = "Theme",
            Options = {"Midnight", "Obsidian", "Frost", "Slate"},
            Default = ThemeManager.Current,
            Flag = "__cac_theme",
            Callback = function(val)
                ThemeManager:SetTheme(val)
            end,
        })
        SaveManager:SetIgnored("__cac_theme", true)

        tab:CreateToggle({
            Name = "Reduced Motion",
            Default = false,
            Flag = "__cac_reduced_motion",
            Callback = function(val)
                AnimManager.ReducedMotion = val
            end,
        })
        SaveManager:SetIgnored("__cac_reduced_motion", true)

        tab:CreateToggle({
            Name = "Compact Mode",
            Default = false,
            Flag = "__cac_compact",
            Callback = function(val)
                AnimManager.CompactMode = val
            end,
        })
        SaveManager:SetIgnored("__cac_compact", true)

        -- Config Section
        tab:CreateSection("Configuration")

        local profileDropdown
        local function refreshProfiles()
            local profiles = SaveManager:ListProfiles()
            if #profiles == 0 then profiles = {"default"} end
            if profileDropdown then
                profileDropdown:SetOptions(profiles)
            end
            return profiles
        end

        profileDropdown = tab:CreateDropdown({
            Name = "Profile",
            Options = refreshProfiles(),
            Default = SaveManager.DefaultProfile,
            Flag = "__cac_profile",
        })
        SaveManager:SetIgnored("__cac_profile", true)

        tab:CreateButton({
            Name = "Save Profile",
            Description = "Save current settings to the selected profile",
            Callback = function()
                local p = SaveManager.Flags["__cac_profile"] or "default"
                SaveManager:Save(p)
                Window:Notify({Title = "Config", Content = "Saved profile: " .. p, Type = "success"})
            end,
        })

        tab:CreateButton({
            Name = "Load Profile",
            Description = "Load settings from the selected profile",
            Callback = function()
                local p = SaveManager.Flags["__cac_profile"] or "default"
                if SaveManager:Load(p) then
                    Window:Notify({Title = "Config", Content = "Loaded profile: " .. p, Type = "success"})
                else
                    Window:Notify({Title = "Config", Content = "Profile not found: " .. p, Type = "error"})
                end
            end,
        })

        tab:CreateButton({
            Name = "Delete Profile",
            Callback = function()
                local p = SaveManager.Flags["__cac_profile"] or "default"
                Window:Confirm({
                    Title = "Delete Profile",
                    Content = "Are you sure you want to delete '" .. p .. "'?",
                    Callback = function(confirmed)
                        if confirmed then
                            SaveManager:Delete(p)
                            refreshProfiles()
                            Window:Notify({Title = "Config", Content = "Deleted: " .. p, Type = "warning"})
                        end
                    end,
                })
            end,
        })

        tab:CreateButton({
            Name = "Refresh Profile List",
            Callback = function()
                refreshProfiles()
                Window:Notify({Title = "Config", Content = "Profiles refreshed", Type = "info"})
            end,
        })

        tab:CreateButton({
            Name = "Reset All Settings",
            Callback = function()
                Window:Confirm({
                    Title = "Reset",
                    Content = "Reset all settings to defaults?",
                    Callback = function(confirmed)
                        if confirmed then
                            SaveManager:ResetConfig()
                            Window:Notify({Title = "Config", Content = "Settings reset to defaults", Type = "warning"})
                        end
                    end,
                })
            end,
        })

        -- About Section
        tab:CreateSection("About")

        tab:CreateParagraph({
            Title = "CAC Ultimate v" .. Library.Version,
            Content = "A premium GUI framework for Roblox.\nBuilt for plugins, tools, and dashboards.\n\nMade by " .. Library.Author,
        })

        return tab
    end

    -- Auto-load default profile if configured
    if Settings.AutoLoad then
        task.defer(function()
            SaveManager:Load(SaveManager.DefaultProfile)
        end)
    end

    table.insert(Library.Windows, Window)
    Library.ActiveWindow = Window
    return Window
end -- end CreateWindow

-- PUBLIC API ACCESSORS

Library.ThemeManager = ThemeManager
Library.SaveManager = SaveManager
Library.AnimManager = AnimManager
Library.Utility = Utility
Library.Flags = SaveManager.Flags

function Library:GetFlag(flag)
    return SaveManager.Flags[flag]
end

function Library:SetFlag(flag, value)
    SaveManager.Flags[flag] = value
    if SaveManager.Options[flag] then
        pcall(function() SaveManager.Options[flag]:SetValue(value) end)
    end
end

function Library:Notify(config)
    if self.ActiveWindow then
        self.ActiveWindow:Notify(config)
    end
end

function Library:Destroy()
    for _, w in ipairs(self.Windows) do
        pcall(function() w:Destroy() end)
    end
    self.Windows = {}
    self.ActiveWindow = nil
end
